83/* Delete a cJSON structure. */
84void cJSON_Delete(cJSON *c)
85{
86 cJSON *next;
87 while (c)
88 {
89 next=c->next;
90 if (!(c->type&cJSON_IsReference) && c->child) cJSON_Delete(c->child);
91 if (!(c->type&cJSON_IsReference) && c->valuestring) cJSON_free(c->valuestring);
92 if (!(c->type&cJSON_StringIsConst) && c->string) cJSON_free(c->string);
93 cJSON_free(c);
94 c=next;
95 }

98/* Parse the input text to generate a number, and populate the result into item. */
99static const char *parse_number(cJSON *item,const char *num)
100{
101 double n=0,sign=1,scale=0;int subscale=0,signsubscale=1;
102
103 if (*num=='-') sign=-1,num++; /* Has sign? */
104 if (*num=='0') num++; /* is zero */
105 if (*num>='1' && *num<='9') do n=(n*10.0)+(*num++ -'0'); while (*num>='0' && *num<='9'); /* Number? */
106 if (*num=='.' && num[1]>='0' && num[1]<='9') {num++; do n=(n*10.0)+(*num++ -'0'),scale--; while (*num>='0' && *num<='9');} /* Fractional part? */
107 if (*num=='e' || *num=='E') /* Exponent? */
108 { num++;if (*num=='+') num++; else if (*num=='-') signsubscale=-1,num++; /* With sign? */
109 while (*num>='0' && *num<='9') subscale=(subscale*10)+(*num++ - '0'); /* Number? */
110 }
111
112 n=sign*n*pow(10.0,(scale+subscale*signsubscale)); /* number = +/- number.fraction * 10^+/- exponent */
113
114 item->valuedouble=n;
115 item->valueint=(int)n;
116 item->type=cJSON_Number;
117 return num;
118}

149/* Render the number nicely from the given item into a string. */
150static char *print_number(cJSON *item,printbuffer *p)
151{
152 char *str=0;
153 double d=item->valuedouble;
154 if (d==0)
155 {
156 if (p) str=ensure(p,2);
157 else str=(char*)cJSON_malloc(2); /* special case for 0. */
158 if (str) strcpy(str,"0");
159 }
160 else if (fabs(((double)item->valueint)-d)<=DBL_EPSILON && d<=INT_MAX && d>=INT_MIN)
161 {
162 if (p) str=ensure(p,21);
163 else str=(char*)cJSON_malloc(21); /* 2^64+1 can be represented in 21 chars. */
164 if (str) sprintf(str,"%d",item->valueint);
165 }
166 else
167 {
168 if (p) str=ensure(p,64);
169 else str=(char*)cJSON_malloc(64); /* This is a nice tradeoff. */
170 if (str)
171 {
172 if (fabs(floor(d)-d)<=DBL_EPSILON && fabs(d)<1.0e60)sprintf(str,"%.0f",d);
173 else if (fabs(d)<1.0e-6 || fabs(d)>1.0e9) sprintf(str,"%e",d);
174 else sprintf(str,"%f",d);
175 }
176 }
177 return str;
178}

194static const unsigned char firstByteMark[7] = { 0x00, 0x00, 0xC0, 0xE0, 0xF0, 0xF8, 0xFC };
195static const char *parse_string(cJSON *item,const char *str)
196{
197 const char *ptr=str+1;char *ptr2;char *out;int len=0;unsigned uc,uc2;
198 if (*str!='\"') {ep=str;return 0;} /* not a string! */
199
200 while (*ptr!='\"' && *ptr && ++len) if (*ptr++ == '\\') ptr++; /* Skip escaped quotes. */
201
202 out=(char*)cJSON_malloc(len+1); /* This is how long we need for the string, roughly. */
203 if (!out) return 0;
204
205 ptr=str+1;ptr2=out;
206 while (*ptr!='\"' && *ptr)
207 {
208 if (*ptr!='\\') *ptr2++=*ptr++;
209 else
210 {
211 ptr++;
212 switch (*ptr)
213 {
214 case 'b': *ptr2++='\b'; break;
215 case 'f': *ptr2++='\f'; break;
216 case 'n': *ptr2++='\n'; break;
217 case 'r': *ptr2++='\r'; break;
218 case 't': *ptr2++='\t'; break;
219 case 'u': /* transcode utf16 to utf8. */
220 uc=parse_hex4(ptr+1);ptr+=4; /* get the unicode char. */
221
222 if ((uc>=0xDC00 && uc<=0xDFFF) || uc==0) break; /* check for invalid. */
223
224 if (uc>=0xD800 && uc<=0xDBFF) /* UTF16 surrogate pairs. */
225 {
226 if (ptr[1]!='\\' || ptr[2]!='u') break; /* missing second-half of surrogate. */
227 uc2=parse_hex4(ptr+3);ptr+=6;
228 if (uc2<0xDC00 || uc2>0xDFFF) break; /* invalid second-half of surrogate. */
229 uc=0x10000 + (((uc&0x3FF)<<10) | (uc2&0x3FF));
230 }
231
232 len=4;if (uc<0x80) len=1;else if (uc<0x800) len=2;else if (uc<0x10000) len=3; ptr2+=len;
233
234 switch (len) {
235 case 4: *--ptr2 =((uc | 0x80) & 0xBF); uc >>= 6;
236 case 3: *--ptr2 =((uc | 0x80) & 0xBF); uc >>= 6;
237 case 2: *--ptr2 =((uc | 0x80) & 0xBF); uc >>= 6;
238 case 1: *--ptr2 =(uc | firstByteMark[len]);
239 }
240 ptr2+=len;
241 break;
242 default: *ptr2++=*ptr; break;
243 }
244 ptr++;
245 }
246 }
247 *ptr2=0;
248 if (*ptr=='\"') ptr++;
249 item->valuestring=out;
250 item->type=cJSON_String;
251 return ptr;
252}

254/* Render the cstring provided to an escaped version that can be printed. */
255static char *print_string_ptr(const char *str,printbuffer *p)
256{
257 const char *ptr;char *ptr2,*out;int len=0,flag=0;unsigned char token;
258
259 for (ptr=str;*ptr;ptr++) flag|=((*ptr>0 && *ptr<32)||(*ptr=='\"')||(*ptr=='\\'))?1:0;
260 if (!flag)
261 {
262 len=ptr-str;
263 if (p) out=ensure(p,len+3);
264 else out=(char*)cJSON_malloc(len+3);
265 if (!out) return 0;
266 ptr2=out;*ptr2++='\"';
267 strcpy(ptr2,str);
268 ptr2[len]='\"';
269 ptr2[len+1]=0;
270 return out;
271 }
272
273 if (!str)
274 {
275 if (p) out=ensure(p,3);
276 else out=(char*)cJSON_malloc(3);
277 if (!out) return 0;
278 strcpy(out,"\"\"");
279 return out;
280 }
281 ptr=str;while ((token=*ptr) && ++len) {if (strchr("\"\\\b\f\n\r\t",token)) len++; else if (token<32) len+=5;ptr++;}
282
283 if (p) out=ensure(p,len+3);
284 else out=(char*)cJSON_malloc(len+3);
285 if (!out) return 0;
286
287 ptr2=out;ptr=str;
288 *ptr2++='\"';
289 while (*ptr)
290 {
291 if ((unsigned char)*ptr>31 && *ptr!='\"' && *ptr!='\\') *ptr2++=*ptr++;
292 else
293 {
294 *ptr2++='\\';
295 switch (token=*ptr++)
296 {
297 case '\\': *ptr2++='\\'; break;
298 case '\"': *ptr2++='\"'; break;
299 case '\b': *ptr2++='b'; break;
300 case '\f': *ptr2++='f'; break;
301 case '\n': *ptr2++='n'; break;
302 case '\r': *ptr2++='r'; break;
303 case '\t': *ptr2++='t'; break;
304 default: sprintf(ptr2,"u%04x",token);ptr2+=5; break; /* escape and print */
305 }
306 }
307 }
308 *ptr2++='\"';*ptr2++=0;
309 return out;
310}

359/* Parser core - when encountering text, process appropriately. */
360static const char *parse_value(cJSON *item,const char *value)
361{
362 if (!value) return 0; /* Fail on null. */
363 if (!strncmp(value,"null",4)) { item->type=cJSON_NULL; return value+4; }
364 if (!strncmp(value,"false",5)) { item->type=cJSON_False; return value+5; }
365 if (!strncmp(value,"true",4)) { item->type=cJSON_True; item->valueint=1; return value+4; }
366 if (*value=='\"') { return parse_string(item,value); }
367 if (*value=='-' || (*value>='0' && *value<='9')) { return parse_number(item,value); }
368 if (*value=='[') { return parse_array(item,value); }
369 if (*value=='{') { return parse_object(item,value); }
370
371 ep=value;return 0; /* failure. */
372}

375static char *print_value(cJSON *item,int depth,int fmt,printbuffer *p)
376{
377 char *out=0;
378 if (!item) return 0;
379 if (p)
380 {
381 switch ((item->type)&255)
382 {
383 case cJSON_NULL: {out=ensure(p,5); if (out) strcpy(out,"null"); break;}
384 case cJSON_False: {out=ensure(p,6); if (out) strcpy(out,"false"); break;}
385 case cJSON_True: {out=ensure(p,5); if (out) strcpy(out,"true"); break;}
386 case cJSON_Number: out=print_number(item,p);break;
387 case cJSON_String: out=print_string(item,p);break;
388 case cJSON_Array: out=print_array(item,depth,fmt,p);break;
389 case cJSON_Object: out=print_object(item,depth,fmt,p);break;
390 }

392 else
393 {
394 switch ((item->type)&255)
395 {
396 case cJSON_NULL: out=cJSON_strdup("null"); break;
397 case cJSON_False: out=cJSON_strdup("false");break;
398 case cJSON_True: out=cJSON_strdup("true"); break;
399 case cJSON_Number: out=print_number(item,0);break;
400 case cJSON_String: out=print_string(item,0);break;

402 case cJSON_Object: out=print_object(item,depth,fmt,0);break;
403 }

408/* Build an array from input text. */
409static const char *parse_array(cJSON *item,const char *value)
410{
411 cJSON *child;
412 if (*value!='[') {ep=value;return 0;} /* not an array! */
413
414 item->type=cJSON_Array;
415 value=skip(value+1);
416 if (*value==']') return value+1; /* empty array. */
417
418 item->child=child=cJSON_New_Item();
419 if (!item->child) return 0; /* memory fail */
420 value=skip(parse_value(child,skip(value))); /* skip any spacing, get the value. */
421 if (!value) return 0;
422
423 while (*value==',')
424 {
425 cJSON *new_item;
426 if (!(new_item=cJSON_New_Item())) return 0; /* memory fail */
427 child->next=new_item;new_item->prev=child;child=new_item;
428 value=skip(parse_value(child,skip(value+1)));
429 if (!value) return 0; /* memory fail */
430 }
431
432 if (*value==']') return value+1; /* end of array */
433 ep=value;return 0; /* malformed. */
434}

436/* Render an array to text */
437static char *print_array(cJSON *item,int depth,int fmt,printbuffer *p)
438{
439 char **entries;
440 char *out=0,*ptr,*ret;int len=5;
441 cJSON *child=item->child;
442 int numentries=0,i=0,fail=0;
443 size_t tmplen=0;
444
445 /* How many entries in the array? */
446 while (child) numentries++,child=child->next;
447 /* Explicitly handle numentries==0 */
448 if (!numentries)
449 {
450 if (p) out=ensure(p,3);
451 else out=(char*)cJSON_malloc(3);
452 if (out) strcpy(out,"[]");
453 return out;
454 }
455
456 if (p)
457 {
458 /* Compose the output array. */
459 i=p->offset;
460 ptr=ensure(p,1);if (!ptr) return 0; *ptr='['; p->offset++;
461 child=item->child;
462 while (child && !fail)
463 {
464 print_value(child,depth+1,fmt,p);
465 p->offset=update(p);
466 if (child->next) {len=fmt?2:1;ptr=ensure(p,len+1);if (!ptr) return 0;*ptr++=',';if(fmt)*ptr++=' ';*ptr=0;p->offset+=len;}
467 child=child->next;
468 }
469 ptr=ensure(p,2);if (!ptr) return 0; *ptr++=']';*ptr=0;
470 out=(p->buffer)+i;
471 }
472 else
473 {
474 /* Allocate an array to hold the values for each */
475 entries=(char**)cJSON_malloc(numentries*sizeof(char*));
476 if (!entries) return 0;
477 memset(entries,0,numentries*sizeof(char*));
478 /* Retrieve all the results: */
479 child=item->child;
480 while (child && !fail)
481 {
482 ret=print_value(child,depth+1,fmt,0);
483 entries[i++]=ret;
484 if (ret) len+=strlen(ret)+2+(fmt?1:0); else fail=1;
485 child=child->next;
486 }
487
488 /* If we didn't fail, try to malloc the output string */
489 if (!fail) out=(char*)cJSON_malloc(len);
490 /* If that fails, we fail. */
491 if (!out) fail=1;
492
493 /* Handle failure. */
494 if (fail)
495 {
496 for (i=0;i<numentries;i++) if (entries[i]) cJSON_free(entries[i]);
497 cJSON_free(entries);
498 return 0;
499 }
500
501 /* Compose the output array. */
502 *out='[';
503 ptr=out+1;*ptr=0;
504 for (i=0;i<numentries;i++)
505 {
506 tmplen=strlen(entries[i]);memcpy(ptr,entries[i],tmplen);ptr+=tmplen;
507 if (i!=numentries-1) {*ptr++=',';if(fmt)*ptr++=' ';*ptr=0;}
508 cJSON_free(entries[i]);
509 }
510 cJSON_free(entries);
511 *ptr++=']';*ptr++=0;
512 }
513 return out;
514}

552/* Render an object to text. */
553static char *print_object(cJSON *item,int depth,int fmt,printbuffer *p)
554{
555 char **entries=0,**names=0;
556 char *out=0,*ptr,*ret,*str;int len=7,i=0,j;
557 cJSON *child=item->child;
558 int numentries=0,fail=0;
559 size_t tmplen=0;
560 /* Count the number of entries. */
561 while (child) numentries++,child=child->next;
562 /* Explicitly handle empty object case */
563 if (!numentries)
564 {
565 if (p) out=ensure(p,fmt?depth+4:3);
566 else out=(char*)cJSON_malloc(fmt?depth+4:3);
567 if (!out) return 0;
568 ptr=out;*ptr++='{';
569 if (fmt) {*ptr++='\n';for (i=0;i<depth-1;i++) *ptr++='\t';}
570 *ptr++='}';*ptr++=0;
571 return out;
572 }
573 if (p)
574 {
575 /* Compose the output: */
576 i=p->offset;
577 len=fmt?2:1; ptr=ensure(p,len+1); if (!ptr) return 0;
578 *ptr++='{'; if (fmt) *ptr++='\n'; *ptr=0; p->offset+=len;
579 child=item->child;depth++;
580 while (child)
581 {
582 if (fmt)
583 {
584 ptr=ensure(p,depth); if (!ptr) return 0;
585 for (j=0;j<depth;j++) *ptr++='\t';
586 p->offset+=depth;
587 }
588 print_string_ptr(child->string,p);
589 p->offset=update(p);
590
591 len=fmt?2:1;
592 ptr=ensure(p,len); if (!ptr) return 0;
593 *ptr++=':';if (fmt) *ptr++='\t';

710/* Duplication */
711cJSON *cJSON_Duplicate(cJSON *item,int recurse)
712{
713 cJSON *newitem,*cptr,*nptr=0,*newchild;
714 /* Bail on bad ptr */
715 if (!item) return 0;
716 /* Create new item */
717 newitem=cJSON_New_Item();
718 if (!newitem) return 0;
719 /* Copy over all vars */
720 newitem->type=item->type&(~cJSON_IsReference),newitem->valueint=item->valueint,newitem->valuedouble=item->valuedouble;
721 if (item->valuestring) {newitem->valuestring=cJSON_strdup(item->valuestring); if (!newitem->valuestring) {cJSON_Delete(newitem);return 0;}}
722 if (item->string) {newitem->string=cJSON_strdup(item->string); if (!newitem->string) {cJSON_Delete(newitem);return 0;}}
723 /* If non-recursive, then we're done! */
724 if (!recurse) return newitem;
725 /* Walk the ->next chain for the child. */
726 cptr=item->child;
727 while (cptr)
728 {
729 newchild=cJSON_Duplicate(cptr,1); /* Duplicate (with recurse) each item in the ->next chain */
730 if (!newchild) {cJSON_Delete(newitem);return 0;}
731 if (nptr) {nptr->next=newchild,newchild->prev=nptr;nptr=newchild;} /* If newitem->child already set, then crosswire ->prev and ->next and move on */
732 else {newitem->child=newchild;nptr=newchild;} /* Set newitem->child and move to it */
733 cptr=cptr->next;
734 }
735 return newitem;
736}
