45static const char * strlwc(const char * in, char *out, unsigned len)
46{
47 unsigned i ;
48
49 if (in==NULL || out == NULL || len==0) return NULL ;
50 i=0 ;
51 while (in[i] != '\0' && i < len-1) {
52 out[i] = (char)tolower((int)in[i]);
53 i++ ;
54 }
55 out[i] = '\0';
56 return out ;
57}

95static unsigned strstrip(char * s)
96{
97 char *last = NULL ;
98 char *dest = s;
99
100 if (s==NULL) return 0;
101
102 last = s + strlen(s);
103 while (isspace((int)*s) && *s) s++;
104 while (last > s) {
105 if (!isspace((int)*(last-1)))
106 break ;
107 last -- ;
108 }
109 *last = (char)0;
110
111 memmove(dest,s,last - s + 1);
112 return last - s;
113}

637static line_status iniparser_line(
638 const char * input_line,
639 char * section,
640 char * key,
641 char * value)
642{
643 line_status sta ;
644 char * line = NULL;
645 size_t len ;
646
647 line = xstrdup(input_line);
648 len = strstrip(line);
649
650 sta = LINE_UNPROCESSED ;
651 if (len<1) {
652 /* Empty line */
653 sta = LINE_EMPTY ;
654 } else if (line[0]=='#' || line[0]==';') {
655 /* Comment line */
656 sta = LINE_COMMENT ;
657 } else if (line[0]=='[' && line[len-1]==']') {
658 /* Section name */
659 sscanf(line, "[%[^]]", section);
660 strstrip(section);
661 strlwc(section, section, len);
662 sta = LINE_SECTION ;
663 } else if (sscanf (line, "%[^=] = \"%[^\"]\"", key, value) == 2
664 || sscanf (line, "%[^=] = '%[^\']'", key, value) == 2) {
665 /* Usual key=value with quotes, with or without comments */
666 strstrip(key);
667 strlwc(key, key, len);
668 /* Don't strip spaces from values surrounded with quotes */
669 sta = LINE_VALUE ;
670 } else if (sscanf (line, "%[^=] = %[^;#]", key, value) == 2) {
671 /* Usual key=value without quotes, with or without comments */
672 strstrip(key);
673 strlwc(key, key, len);
674 strstrip(value);
675 /*
676 * sscanf cannot handle '' or "" as empty values
677 * this is done here
678 */
679 if (!strcmp(value, "\"\"") || (!strcmp(value, "''"))) {
680 value[0]=0 ;
681 }
682 sta = LINE_VALUE ;
683 } else if (sscanf(line, "%[^=] = %[;#]", key, value)==2
684 || sscanf(line, "%[^=] %[=]", key, value) == 2) {
685 /*
686 * Special cases:
687 * key=
688 * key=;
689 * key=#
690 */
691 strstrip(key);
692 strlwc(key, key, len);
693 value[0]=0 ;
694 sta = LINE_VALUE ;
695 } else {
696 /* Generate syntax error */
697 sta = LINE_ERROR ;
698 }
699
700 free(line);
701 return sta ;
702}

717/*--------------------------------------------------------------------------*/
718dictionary * iniparser_load(const char * ininame)
719{
720 FILE * in ;
721
722 char line [ASCIILINESZ+1] ;
723 char section [ASCIILINESZ+1] ;
724 char key [ASCIILINESZ+1] ;
725 char tmp [(ASCIILINESZ * 2) + 2] ;
726 char val [ASCIILINESZ+1] ;
727
728 int last=0 ;
729 int len ;
730 int lineno=0 ;
731 int errs=0;
732 int mem_err=0;
733
734 dictionary * dict ;
735
736 if ((in=fopen(ininame, "r"))==NULL) {
737 iniparser_error_callback("iniparser: cannot open %s\n", ininame);
738 return NULL ;
739 }
740
741 dict = dictionary_new(0) ;
742 if (!dict) {
743 fclose(in);
744 return NULL ;
745 }
746
747 memset(line, 0, ASCIILINESZ);
748 memset(section, 0, ASCIILINESZ);
749 memset(key, 0, ASCIILINESZ);
750 memset(val, 0, ASCIILINESZ);
751 last=0 ;
752
753 while (fgets(line+last, ASCIILINESZ-last, in)!=NULL) {
754 lineno++ ;
755 len = (int)strlen(line)-1;
756 if (len<=0)
757 continue;
758 /* Safety check against buffer overflows */
759 if (line[len]!='\n' && !feof(in)) {
760 iniparser_error_callback(
761 "iniparser: input line too long in %s (%d)\n",
762 ininame,
763 lineno);
764 dictionary_del(dict);
765 fclose(in);
766 return NULL ;
767 }
768 /* Get rid of \n and spaces at end of line */
769 while ((len>=0) &&
770 ((line[len]=='\n') || (isspace(line[len])))) {
771 line[len]=0 ;
772 len-- ;
773 }
774 if (len < 0) { /* Line was entirely \n and/or spaces */
775 len = 0;
776 }
777 /* Detect multi-line */
778 if (line[len]=='\\') {
779 /* Multi-line value */
780 last=len ;
781 continue ;
782 } else {
783 last=0 ;
784 }
785 switch (iniparser_line(line, section, key, val)) {
786 case LINE_EMPTY:
787 case LINE_COMMENT:
788 break ;
789
790 case LINE_SECTION:
791 mem_err = dictionary_set(dict, section, NULL);
792 break ;
793
794 case LINE_VALUE:
795 sprintf(tmp, "%s:%s", section, key);
796 mem_err = dictionary_set(dict, tmp, val);
797 break ;
798
799 case LINE_ERROR:
800 iniparser_error_callback(
801 "iniparser: syntax error in %s (%d):\n-> %s\n",
802 ininame,
803 lineno,
804 line);
805 errs++ ;
806 break;
807
808 default:
809 break ;
810 }
811 memset(line, 0, ASCIILINESZ);
812 last=0;
813 if (mem_err<0) {
814 iniparser_error_callback("iniparser: memory allocation failure\n");
815 break ;
816 }
817 }
818 if (errs) {
819 dictionary_del(dict);
820 dict = NULL ;
821 }
822 fclose(in);
823 return dict ;
824}
